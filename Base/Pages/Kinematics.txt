import { useState } from "react";
import { useQuery } from "@tanstack/react-query";
import { base44 } from "@/api/base44Client";
import { Link } from "react-router-dom";
import { createPageUrl } from "@/utils";
import { Card } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { Badge } from "@/components/ui/badge";
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table";
import { ArrowLeft, Navigation, Clock, Ruler, Download, Filter } from "lucide-react";
import { format } from "date-fns";
import { cn } from "@/lib/utils";
import PathMap from "@/components/path/PathMap";
import JointVisualizer from "@/components/robot/JointVisualizer";
import JointDetailPanel from "@/components/robot/JointDetailPanel";

export default function Kinematics() {
  const [selectedRobotId, setSelectedRobotId] = useState("");
  const [selectedPath, setSelectedPath] = useState(null);
  const [selectedJoint, setSelectedJoint] = useState(null);

  const { data: robots } = useQuery({
    queryKey: ['robots'],
    queryFn: () => base44.entities.Robot.list()
  });

  const { data: pathLogs } = useQuery({
    queryKey: ['paths', selectedRobotId],
    queryFn: () => selectedRobotId 
      ? base44.entities.PathLog.filter({ robot_id: selectedRobotId }, '-created_date', 50)
      : base44.entities.PathLog.list('-created_date', 50),
    enabled: true
  });

  const { data: joints } = useQuery({
    queryKey: ['joints', selectedRobotId],
    queryFn: () => base44.entities.JointData.filter({ robot_id: selectedRobotId }),
    enabled: !!selectedRobotId
  });

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-50 via-indigo-50/30 to-slate-100 p-4 md:p-6">
      <div className="max-w-7xl mx-auto">
        {/* Header */}
        <div className="flex flex-col md:flex-row md:items-center justify-between gap-4 mb-6">
          <div className="flex items-center gap-4">
            <Link to={createPageUrl("Dashboard")}>
              <Button variant="ghost" size="icon">
                <ArrowLeft className="w-5 h-5" />
              </Button>
            </Link>
            <div>
              <h1 className="text-2xl font-bold text-slate-800">Kinematics & Path Logging</h1>
              <p className="text-slate-500 text-sm">Joint assessment and movement history</p>
            </div>
          </div>
          
          <div className="flex items-center gap-3">
            <Select value={selectedRobotId} onValueChange={setSelectedRobotId}>
              <SelectTrigger className="w-48">
                <SelectValue placeholder="All Robots" />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value={null}>All Robots</SelectItem>
                {robots?.map(robot => (
                  <SelectItem key={robot.id} value={robot.id}>
                    {robot.name}
                  </SelectItem>
                ))}
              </SelectContent>
            </Select>
            <Button variant="outline">
              <Download className="w-4 h-4 mr-2" />
              Export
            </Button>
          </div>
        </div>

        <div className="grid grid-cols-1 xl:grid-cols-3 gap-6">
          {/* Path Map */}
          <div className="xl:col-span-2">
            <PathMap pathLog={pathLogs} selectedPath={selectedPath} />
          </div>

          {/* Joint Status */}
          <div>
            {selectedRobotId && joints?.length > 0 ? (
              <JointVisualizer joints={joints} onJointSelect={setSelectedJoint} />
            ) : (
              <Card className="p-6 border-0 bg-white/80 h-full flex items-center justify-center">
                <div className="text-center text-slate-400">
                  <Navigation className="w-8 h-8 mx-auto mb-2 opacity-50" />
                  <p className="text-sm">Select a robot to view joint status</p>
                </div>
              </Card>
            )}
          </div>
        </div>

        {/* Selected Joint Detail */}
        {selectedJoint && (
          <div className="mt-6">
            <JointDetailPanel 
              joint={selectedJoint}
              onClose={() => setSelectedJoint(null)}
            />
          </div>
        )}

        {/* Path Log Table */}
        <Card className="mt-6 border-0 bg-white/80 backdrop-blur-sm overflow-hidden">
          <div className="p-4 border-b border-slate-100">
            <h3 className="font-semibold text-slate-800">Path History</h3>
          </div>
          <div className="overflow-x-auto">
            <Table>
              <TableHeader>
                <TableRow className="bg-slate-50/50">
                  <TableHead>Session</TableHead>
                  <TableHead>Robot</TableHead>
                  <TableHead>Start Time</TableHead>
                  <TableHead>Duration</TableHead>
                  <TableHead>Distance</TableHead>
                  <TableHead>Waypoints</TableHead>
                  <TableHead></TableHead>
                </TableRow>
              </TableHeader>
              <TableBody>
                {pathLogs?.length > 0 ? (
                  pathLogs.map((path) => {
                    const robot = robots?.find(r => r.id === path.robot_id);
                    return (
                      <TableRow 
                        key={path.id}
                        className={cn(
                          "cursor-pointer hover:bg-indigo-50/50 transition-colors",
                          selectedPath?.id === path.id && "bg-indigo-50"
                        )}
                        onClick={() => setSelectedPath(path)}
                      >
                        <TableCell>
                          <Badge variant="outline" className="font-mono text-xs">
                            {path.session_id?.slice(0, 8) || 'N/A'}
                          </Badge>
                        </TableCell>
                        <TableCell className="font-medium">{robot?.name || 'Unknown'}</TableCell>
                        <TableCell className="text-slate-500">
                          {path.start_time ? format(new Date(path.start_time), 'MMM dd, HH:mm') : '-'}
                        </TableCell>
                        <TableCell>
                          <div className="flex items-center gap-1 text-slate-600">
                            <Clock className="w-3 h-3" />
                            {path.duration_seconds ? `${Math.floor(path.duration_seconds / 60)}m ${path.duration_seconds % 60}s` : '-'}
                          </div>
                        </TableCell>
                        <TableCell>
                          <div className="flex items-center gap-1 text-slate-600">
                            <Ruler className="w-3 h-3" />
                            {path.total_distance?.toFixed(1) || 0}m
                          </div>
                        </TableCell>
                        <TableCell>
                          <Badge className="bg-indigo-100 text-indigo-700 border-0">
                            {path.waypoints?.length || 0}
                          </Badge>
                        </TableCell>
                        <TableCell>
                          <Button variant="ghost" size="sm" className="text-indigo-600">
                            View
                          </Button>
                        </TableCell>
                      </TableRow>
                    );
                  })
                ) : (
                  <TableRow>
                    <TableCell colSpan={7} className="text-center py-8 text-slate-400">
                      No path logs available
                    </TableCell>
                  </TableRow>
                )}
              </TableBody>
            </Table>
          </div>
        </Card>
      </div>
    </div>
  );
}
